import pyodbc

server = 'MINHLAP\SQLEXPRESS'
conn = pyodbc.connect('DRIVER={SQL Server};SERVER='+server+';DATABASE=AdventureWorks2019;Trusted_Connection=yes;')
cursor = conn.cursor()

# query data t  preson.EmailAddress, preson.BusinessEntityAddress, v preson.Address
cursor.execute('''
        SELECT a.AddressID,   
            e.EmailAddress,
            a.AddressLine1,
            a.AddressLine2,
            a.StateProvinceID,
            a.City,
            a.PostalCode,
            a.SpatialLocation,
            a.rowguid,
            a.ModifiedDate
        FROM Person.EmailAddress as e
        JOIN Person.BusinessEntityAddress as bea
            ON e.BusinessEntityID = bea.BusinessEntityID
        JOIN Person.Address as a
            ON bea.AddressID = a.AddressID       
    ''')
# lu tt c data vo  dataframe
rows = cursor.fetchall()

columns = [column[0] for column in cursor.description]

df = pd.DataFrame.from_records(rows, columns=columns)
